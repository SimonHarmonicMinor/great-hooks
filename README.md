## Install

```bash
npm install --save great-hooks
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'great-hooks'
import 'great-hooks/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
